1. Reverse a string using a stack
function reverseString(word) {
  let stack = [];
  for (let i = 0; i < word.length; i++) {
    stack.push(word[i]);
  }
  let reversedWord = '';
  for (let i = stack.length - 1; i >= 0; i--) {
    reversedWord+= stack[i];
  }
  return reversedWord;
}

3. Implement enqueue and dequeue in your own Queue class
class Queue {
  constructor(list, element) {
    this.list = list;
    this.element = element;
  }
  enqueue() {
    this.list[this.list.length] = this.element;
  }
  dequeue() {
    let list = this.list;
    let temp = list[0];
    for (let i = 1; i < list.length; i++) {
      list[i - 1] = list[i];
    }
    list.pop();
    return temp;
  }
}
